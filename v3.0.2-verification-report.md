# Log filtering results
original_count=$(echo "$collected_files" | grep -v '^\s*$' | wc -l | xargs)
filtered_count=$(echo "$filtered_files" | grep -v '^\s*$' | wc -l | xargs)
echo "🔍 Filtered files: ${original_count} → ${filtered_count}"
